60  /**
61   * Versionalize Java packages.
62   *
63   * @author Yegor Bugayenko (yegor@tpc2.com)
64   * @version $Id: 163f4a824d25af7718c40c4310068aa8f0b917b7 $
65   * @since 0.7.16
66   */
67  @Mojo(
68      name = "versionalize",
69      defaultPhase = LifecyclePhase.PREPARE_PACKAGE,
70      threadSafe = true
71  )
72  @ToString
73  @EqualsAndHashCode(callSuper = false)
74  @Loggable(Loggable.DEBUG)
75  @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
76  public final class VersionalizeMojo extends AbstractMojo {
77  
78      /**
79       * Maven project.
80       */
81      @Component
82      private transient MavenProject project;
83  
84      /**
85       * Build number.
86       * @checkstyle MemberNameCheck (10 lines)
87       */
88      @Parameter(
89          property = "buildNumber",
90          required = false,
91          readonly = false
92      )
93      private transient String buildNumber;
94  
95      @Override
96      @Loggable(value = Loggable.DEBUG, limit = 1, unit = TimeUnit.MINUTES)
97      public void execute() throws MojoFailureException {
98          StaticLoggerBinder.getSingleton().setMavenLog(this.getLog());
99          final File src = new File(this.project.getBuild().getSourceDirectory());
100         if (!src.exists()) {
101             Logger.info(this, "source directory '%s' is absent", src);
102             return;
103         }
104         final File dest =
105             new File(this.project.getBuild().getOutputDirectory());
106         if (dest.mkdirs()) {
107             Logger.info(this, "created directory %s", dest);
108         }
109         Logger.info(this, "Versionalizing %s directory", dest);
110         try {
111             this.versionalize(src, dest);
112         } catch (final IOException ex) {
113             throw new MojoFailureException("failed to versionalize", ex);
114         }
115     }
116 
117     /**
118      * Create and return a text of the version file.
119      * @param dir The destination directory
120      * @return The text
121      */
122     private String text(@NotNull final File dir) {
123         final StringBuilder text = new StringBuilder(0)
124             .append(String.format("Build Number: %s%n", this.buildNumber))
125             .append(
126                 String.format(
127                     "Project Version: %s%n",
128                     this.project.getVersion()
129                 )
130             )
131             .append(
132                 String.format(
133                     "Build Date: %s%n%n",
134                     DateFormatUtils.ISO_DATETIME_FORMAT.format(new Date())
135                 )
136             );
137         for (final String name : VersionalizeMojo.files(dir, "*")) {
138             final File file = new File(dir, name);
139             if (file.isFile()) {
140                 text.append(name)
141                     .append(": ")
142                     .append(file.length())
143                     .append('\n');
144             }
145         }
146         return text.toString();
147     }
148 
149     /**
150      * Versionalize packages from source to dest.
151      * @param src Source directory
152      * @param dest Destination
153      * @throws IOException If some IO problem
154      */
155     private void versionalize(@NotNull final File src, @NotNull final File dest)
156         throws IOException {
157         final Collection<File> dirs = FileUtils.listFilesAndDirs(
158             src,
159             new NotFileFilter(TrueFileFilter.INSTANCE),
160             DirectoryFileFilter.DIRECTORY
161         );
162         final String name = String.format(
163             "%s-%s-%s.txt",
164             VersionalizeMojo.cleanup(this.project.getGroupId()),
165             VersionalizeMojo.cleanup(this.project.getArtifactId()),
166             VersionalizeMojo.cleanup(this.project.getPackaging())
167         );
168         for (final File dir : dirs) {
169             if (VersionalizeMojo.files(dir, "*.java").isEmpty()) {
170                 continue;
171             }
172             final File ddir = new File(
173                 dest,
174                 StringUtils.substring(
175                     dir.getCanonicalPath(),
176                     src.getCanonicalPath().length() + 1
177                 )
178             );
179             final File version = new File(ddir, name);
180             if (version.getParentFile().mkdirs()) {
181                 Logger.info(this, "created dir %s", version.getParentFile());
182             }
183             FileUtils.write(version, this.text(ddir));
184             Logger.info(this, "File %s added", version);
185         }
186     }
187 
188     /**
189      * Clean the text.
190      * @param text The text
191      * @return Clean version of it
192      */
193     private static String cleanup(final String text) {
194         return text.replaceAll("[^_a-z0-9\\-]", "-");
195     }
196 
197     /**
198      * All Java files in the directory.
199      * @param dir The directory
200      * @param mask Mask to use
201      * @return List of Java file names
202      */
203     private static Collection<String> files(final File dir, final String mask) {
204         final FileFilter filter = new WildcardFileFilter(mask);
205         final File[] files = dir.listFiles(filter);
206         final Collection<String> names = new ArrayList<String>(files.length);
207         for (final File file : files) {
208             names.add(file.getName());
209         }
210         return names;
211     }
